Recipe 06 — Uniforms & Transforms

Move geometry with matrices. setTransform for the model, setViewTransform for the camera, and bgfx's free u_modelViewProj — plus your own custom uniform.

What you'll learn

  • The bgfx MVP shortcut: set model + view + projection matrices in C++, read the combined u_modelViewProj in the shader — no push constants.
  • Building matrices with bx::mtxLookAt, bx::mtxProj and the rotate helpers in <bx/math.h>.
  • Handing each backend the correct clip-space depth range via bgfx::getCaps()->homogeneousDepth.
  • Declaring a custom uniform with createUniform and feeding it with setUniform.

Prerequisites

The static quad from recipe 05, and the shaderc workflow from recipe 04 — we author a new shader pair here, vs_transform / fs_transform.

Walkthrough

bgfx does the matrix multiply for you

In the 3D track you packed a model-view-projection matrix into a push constant by hand. bgfx takes a different route: you give it the three matrices separately and it exposes the products as predefined uniforms — u_model, u_view, u_proj, u_viewProj and the one we want, u_modelViewProj. The vertex shader just reads it: 

# vs_transform.sc
$input  a_position, a_color0
$output v_color0
#include "bgfx_shader.sh"
void main() {
    gl_Position = mul(u_modelViewProj, vec4(a_position, 1.0));   // mul(), not *
    v_color0    = a_color0;
}

A custom uniform

Built-ins cover transforms; anything else you declare yourself. Here a u_tint colour multiplies the fragment output. Remember the dialect rule from recipe 04: uniforms must be float-based, so a colour is a vec4. 

# fs_transform.sc
$input v_color0
#include "bgfx_shader.sh"
uniform vec4 u_tint;
void main() {
    gl_FragColor = v_color0 * u_tint;
}

On the C++ side you create the uniform once (the name must match the shader) and set its value per frame: 

st->u_tint = bgfx::createUniform("u_tint", bgfx::UniformType::Vec4);   // in init
// ...per frame:
float tint[4] = { 0.6f + 0.4f * sinf(t), 0.8f, 1.0f, 1.0f };
bgfx::setUniform(st->u_tint, tint);

Camera: view and projection

setViewTransform attaches a view matrix and a projection matrix to a view; everything submitted into that view is transformed by them. We build the view with bx::mtxLookAt (camera 5 units back, looking at the origin) and the projection with bx::mtxProj. 

const bx::Vec3 at  = { 0.0f, 0.0f,  0.0f };
const bx::Vec3 eye = { 0.0f, 0.0f, -5.0f };
float view[16], proj[16];
bx::mtxLookAt(view, eye, at);
bx::mtxProj(proj, 60.0f, float(st->width) / float(st->height), 0.1f, 100.0f,
            bgfx::getCaps()->homogeneousDepth);
bgfx::setViewTransform(0, view, proj);
Depth range differs by backend. OpenGL's clip-space Z runs −1…1; Metal, Direct3D and Vulkan use 0…1. Passing bgfx::getCaps()->homogeneousDepth to bx::mtxProj builds the right projection for the active backend — get it wrong and depth testing (recipe 08) inverts.

The model matrix

setTransform sets the model matrix for the next submit. We spin the quad on two axes so you can see the projection at work; with no depth buffer yet it's a single quad, so overlap doesn't matter (recipe 08 fixes that). 

float model[16];
bx::mtxRotateXY(model, t * 0.37f, t * 0.71f);
bgfx::setTransform(model);
bgfx::setVertexBuffer(0, st->vbh);
bgfx::setIndexBuffer(st->ibh);
bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);
bgfx::submit(0, st->program);

Complete listing

/*
 * bgfx/06-uniforms-and-transforms.cpp — a spinning, tinted quad.
 * Native-handle + loadProgram helpers omitted here for length; they are
 * byte-for-byte identical to recipe 03/05. Uses shaders vs_transform/fs_transform.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include <bgfx/bgfx.h>
#include <bgfx/platform.h>
#include <bx/math.h>
#include <cmath>
#include <cstdint>

// nativeWindowHandle / nativeDisplayHandle / loadMem / shaderDir / loadProgram
// — see recipe 03. Elided here so the new logic stands out.
static void *nativeWindowHandle(SDL_Window *);
static void *nativeDisplayHandle(SDL_Window *);
static bgfx::ProgramHandle loadProgram(const char *vs, const char *fs);

struct PosColorVertex { float x, y, z; uint32_t abgr; };
static const PosColorVertex kQuad[4] = {
    { -0.6f,  0.6f, 0.0f, 0xff0000ff },
    {  0.6f,  0.6f, 0.0f, 0xff00ff00 },
    {  0.6f, -0.6f, 0.0f, 0xffff0000 },
    { -0.6f, -0.6f, 0.0f, 0xff00ffff },
};
static const uint16_t kIndices[6] = { 0, 1, 2,  0, 2, 3 };

struct AppState {
    SDL_Window              *window = nullptr;
    int                      width = 1280, height = 720;
    bgfx::VertexLayout       layout;
    bgfx::VertexBufferHandle vbh = BGFX_INVALID_HANDLE;
    bgfx::IndexBufferHandle  ibh = BGFX_INVALID_HANDLE;
    bgfx::ProgramHandle      program = BGFX_INVALID_HANDLE;
    bgfx::UniformHandle      u_tint  = BGFX_INVALID_HANDLE;
    Uint64                   startNs = 0;
};

static SDL_AppResult fail(const char *what) { SDL_Log("%s: %s", what, SDL_GetError()); return SDL_APP_FAILURE; }

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("bgfx Transforms", "1.0", "com.example.sdl3.bgfx.xform");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("SDL_Init");

    AppState *st = new AppState(); *appstate = st;
    st->window = SDL_CreateWindow("bgfx — Uniforms & Transforms", st->width, st->height,
                                  SDL_WINDOW_RESIZABLE | SDL_WINDOW_HIGH_PIXEL_DENSITY);
    if (!st->window) return fail("SDL_CreateWindow");
    SDL_GetWindowSizeInPixels(st->window, &st->width, &st->height);

    bgfx::renderFrame();
    bgfx::Init init;
    init.type              = bgfx::RendererType::Count;
    init.platformData.nwh  = nativeWindowHandle(st->window);
    init.platformData.ndt  = nativeDisplayHandle(st->window);
    init.resolution.width  = (uint32_t)st->width;
    init.resolution.height = (uint32_t)st->height;
    init.resolution.reset  = BGFX_RESET_VSYNC;
    if (!bgfx::init(init)) return fail("bgfx::init");

    bgfx::setViewClear(0, BGFX_CLEAR_COLOR, 0x121a30ff, 1.0f, 0);

    st->layout.begin()
        .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
        .add(bgfx::Attrib::Color0,   4, bgfx::AttribType::Uint8, true)
        .end();
    st->vbh     = bgfx::createVertexBuffer(bgfx::makeRef(kQuad, sizeof(kQuad)), st->layout);
    st->ibh     = bgfx::createIndexBuffer (bgfx::makeRef(kIndices, sizeof(kIndices)));
    st->program = loadProgram("vs_transform", "fs_transform");
    st->u_tint  = bgfx::createUniform("u_tint", bgfx::UniformType::Vec4);
    if (!bgfx::isValid(st->program)) return fail("loadProgram");

    st->startNs = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_WINDOW_PIXEL_SIZE_CHANGED) {
        st->width = e->window.data1; st->height = e->window.data2;
        bgfx::reset((uint32_t)st->width, (uint32_t)st->height, BGFX_RESET_VSYNC);
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (float)(SDL_GetTicksNS() - st->startNs) / 1.0e9f;

    bgfx::setViewRect(0, 0, 0, (uint16_t)st->width, (uint16_t)st->height);

    const bx::Vec3 at  = { 0.0f, 0.0f,  0.0f };
    const bx::Vec3 eye = { 0.0f, 0.0f, -5.0f };
    float view[16], proj[16];
    bx::mtxLookAt(view, eye, at);
    bx::mtxProj(proj, 60.0f, float(st->width) / float(st->height), 0.1f, 100.0f,
                bgfx::getCaps()->homogeneousDepth);
    bgfx::setViewTransform(0, view, proj);

    float model[16];
    bx::mtxRotateXY(model, t * 0.37f, t * 0.71f);
    bgfx::setTransform(model);

    float tint[4] = { 0.6f + 0.4f * sinf(t), 0.8f, 1.0f, 1.0f };
    bgfx::setUniform(st->u_tint, tint);

    bgfx::setVertexBuffer(0, st->vbh);
    bgfx::setIndexBuffer(st->ibh);
    bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);
    bgfx::submit(0, st->program);

    bgfx::frame();
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (bgfx::isValid(st->u_tint))  bgfx::destroy(st->u_tint);
    if (bgfx::isValid(st->ibh))     bgfx::destroy(st->ibh);
    if (bgfx::isValid(st->vbh))     bgfx::destroy(st->vbh);
    if (bgfx::isValid(st->program)) bgfx::destroy(st->program);
    bgfx::shutdown();
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Move the camera: change eye to { 2.0f, 1.0f, -5.0f }. The quad shifts and skews — that's the view matrix, not the model.
  2. Narrow the field of view in bx::mtxProj from 60° to 30°. The quad appears larger and flatter, like zooming a lens.
  3. Replace mtxRotateXY with bx::mtxTranslate(model, sinf(t)*2.0f, 0, 0) to slide it side to side instead of spinning.
  4. Set u_tint to {0,0,0,1}. The quad goes black — every vertex colour is multiplied to zero, proving the custom uniform reaches the fragment shader.
