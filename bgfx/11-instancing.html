Recipe 11 — Instancing

A whole grid of spinning cubes in one submit. Per-instance model matrices ride in an instance data buffer; the GPU stamps out the copies.

What you'll learn

  • Why one submit per object doesn't scale, and what instancing buys you.
  • Filling an instance data buffer with allocInstanceDataBuffer — a model matrix per instance.
  • Checking BGFX_CAPS_INSTANCING and reserving with getAvailInstanceDataBuffer.
  • Reading instance attributes (i_data0…i_data3) in the shader and rebuilding the matrix with mtxFromCols.
  • Why instanced shaders use u_viewProj, not u_modelViewProj.

Prerequisites

The depth-tested cube from recipe 08 and transforms from recipe 06. New vertex shader vs_instanced; the fragment stage reuses fs_cubes.

Walkthrough

One submit, many cubes

The obvious way to draw 121 cubes is a loop of 121 setTransform + submit pairs — 121 draw calls, most of the cost being the per-call overhead, not the triangles. Instancing collapses that: you hand bgfx the shared mesh once plus a buffer of per-instance data, and a single submit draws them all.

The instance data buffer

Per-instance data is just bytes with a stride (a multiple of 16). We store a 4×4 model matrix per cube — 64 bytes. allocInstanceDataBuffer hands you a writable block sized for num × stride; you fill each slot, then bind it. 

const uint16_t kStride = 64;          // one mat4 per instance
const uint32_t kSide   = 11;
const uint32_t kCount  = kSide * kSide;

if (bgfx::getAvailInstanceDataBuffer(kCount, kStride) == kCount) {
    bgfx::InstanceDataBuffer idb;
    bgfx::allocInstanceDataBuffer(&idb, kCount, kStride);

    uint8_t *cursor = idb.data;
    for (uint32_t yy = 0; yy < kSide; ++yy)
        for (uint32_t xx = 0; xx < kSide; ++xx) {
            float *mtx = (float *)cursor;
            bx::mtxRotateXY(mtx, t + xx * 0.21f, t + yy * 0.37f);
            mtx[12] = -15.0f + float(xx) * 3.0f;   // translation X
            mtx[13] = -15.0f + float(yy) * 3.0f;   // translation Y
            mtx[14] = 0.0f;                        // translation Z
            cursor += kStride;
        }

    bgfx::setVertexBuffer(0, st->cubeVbh);
    bgfx::setIndexBuffer(st->cubeIbh);
    bgfx::setInstanceDataBuffer(&idb);
    bgfx::setState(BGFX_STATE_DEFAULT);
    bgfx::submit(0, st->program);                   // ONE draw call for all 121
}

The getAvailInstanceDataBuffer guard matters: the buffer is carved from per-frame scratch, so on a tight frame you might get fewer slots than asked. Drawing what's available beats overrunning.

Instance attributes in the shader

bgfx feeds each instance's bytes to the vertex shader as attributes i_data0…i_data3 — four vec4s, i.e. the matrix columns. Declare them in varying.def.sc: 

# varying.def.sc (instancing)
vec4 v_color0  : COLOR0;

vec3 a_position : POSITION;
vec4 a_color0   : COLOR0;
vec4 i_data0    : TEXCOORD7;     // model matrix, column 0
vec4 i_data1    : TEXCOORD6;     // column 1
vec4 i_data2    : TEXCOORD5;     // column 2
vec4 i_data3    : TEXCOORD4;     // column 3

# vs_instanced.sc
$input  a_position, a_color0, i_data0, i_data1, i_data2, i_data3
$output v_color0
#include "bgfx_shader.sh"
void main() {
    mat4 model  = mtxFromCols(i_data0, i_data1, i_data2, i_data3);
    vec4 world  = mul(model, vec4(a_position, 1.0));
    gl_Position = mul(u_viewProj, world);     // u_viewProj, NOT u_modelViewProj
    v_color0    = a_color0;
}
Why u_viewProj? u_modelViewProj folds in the model matrix set by setTransform — but with instancing there's no single model matrix; each instance brings its own. So we transform by the per-instance model ourselves, then by the shared u_viewProj (which setViewTransform still provides).

Complete listing

Helpers and kCube/kCubeIdx are unchanged from recipes 03 and 07 (elided). If BGFX_CAPS_INSTANCING is missing — rare on modern hardware — you'd fall back to the per-cube submit loop. 

/*
 * bgfx/11-instancing.cpp — an 11x11 grid of spinning cubes in one submit.
 * Shaders: vs_instanced / fs_cubes.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include <bgfx/bgfx.h>
#include <bgfx/platform.h>
#include <bx/math.h>
#include <cstdint>

static void *nativeWindowHandle(SDL_Window *);
static void *nativeDisplayHandle(SDL_Window *);
static bgfx::ProgramHandle loadProgram(const char *vs, const char *fs);

struct PosColorVertex { float x, y, z; uint32_t abgr; };
extern const PosColorVertex kCube[8];     // recipe 07
extern const uint16_t       kCubeIdx[36];

struct AppState {
    SDL_Window              *window = nullptr;
    int                      width = 1280, height = 720;
    bgfx::VertexLayout       layout;
    bgfx::VertexBufferHandle cubeVbh = BGFX_INVALID_HANDLE;
    bgfx::IndexBufferHandle  cubeIbh = BGFX_INVALID_HANDLE;
    bgfx::ProgramHandle      program = BGFX_INVALID_HANDLE;
    Uint64                   startNs = 0;
};

static SDL_AppResult fail(const char *what) { SDL_Log("%s: %s", what, SDL_GetError()); return SDL_APP_FAILURE; }

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("bgfx Instancing", "1.0", "com.example.sdl3.bgfx.inst");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("SDL_Init");

    AppState *st = new AppState(); *appstate = st;
    st->window = SDL_CreateWindow("bgfx — Instancing", st->width, st->height,
                                  SDL_WINDOW_RESIZABLE | SDL_WINDOW_HIGH_PIXEL_DENSITY);
    if (!st->window) return fail("SDL_CreateWindow");
    SDL_GetWindowSizeInPixels(st->window, &st->width, &st->height);

    bgfx::renderFrame();
    bgfx::Init init;
    init.type              = bgfx::RendererType::Count;
    init.platformData.nwh  = nativeWindowHandle(st->window);
    init.platformData.ndt  = nativeDisplayHandle(st->window);
    init.resolution.width  = (uint32_t)st->width;
    init.resolution.height = (uint32_t)st->height;
    init.resolution.reset  = BGFX_RESET_VSYNC;
    if (!bgfx::init(init)) return fail("bgfx::init");

    if (!(bgfx::getCaps()->supported & BGFX_CAPS_INSTANCING))
        SDL_Log("warning: instancing not supported; would need a per-cube fallback");

    bgfx::setViewClear(0, BGFX_CLEAR_COLOR | BGFX_CLEAR_DEPTH, 0x121a30ff, 1.0f, 0);

    st->layout.begin()
        .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
        .add(bgfx::Attrib::Color0,   4, bgfx::AttribType::Uint8, true).end();
    st->cubeVbh = bgfx::createVertexBuffer(bgfx::makeRef(kCube, sizeof(PosColorVertex) * 8), st->layout);
    st->cubeIbh = bgfx::createIndexBuffer (bgfx::makeRef(kCubeIdx, sizeof(uint16_t) * 36));
    st->program = loadProgram("vs_instanced", "fs_cubes");
    if (!bgfx::isValid(st->program)) return fail("loadProgram");

    st->startNs = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_WINDOW_PIXEL_SIZE_CHANGED) {
        st->width = e->window.data1; st->height = e->window.data2;
        bgfx::reset((uint32_t)st->width, (uint32_t)st->height, BGFX_RESET_VSYNC);
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (float)(SDL_GetTicksNS() - st->startNs) / 1.0e9f;

    bgfx::setViewRect(0, 0, 0, (uint16_t)st->width, (uint16_t)st->height);
    const bx::Vec3 at  = { 0.0f,  0.0f,   0.0f };
    const bx::Vec3 eye = { 0.0f,  0.0f, -45.0f };
    float view[16], proj[16];
    bx::mtxLookAt(view, eye, at);
    bx::mtxProj(proj, 60.0f, float(st->width)/float(st->height), 0.1f, 200.0f,
                bgfx::getCaps()->homogeneousDepth);
    bgfx::setViewTransform(0, view, proj);

    const uint16_t kStride = 64;
    const uint32_t kSide   = 11, kCount = kSide * kSide;
    if (bgfx::getAvailInstanceDataBuffer(kCount, kStride) == kCount) {
        bgfx::InstanceDataBuffer idb;
        bgfx::allocInstanceDataBuffer(&idb, kCount, kStride);
        uint8_t *cursor = idb.data;
        for (uint32_t yy = 0; yy < kSide; ++yy)
            for (uint32_t xx = 0; xx < kSide; ++xx) {
                float *mtx = (float *)cursor;
                bx::mtxRotateXY(mtx, t + xx * 0.21f, t + yy * 0.37f);
                mtx[12] = -15.0f + float(xx) * 3.0f;
                mtx[13] = -15.0f + float(yy) * 3.0f;
                mtx[14] = 0.0f;
                cursor += kStride;
            }
        bgfx::setVertexBuffer(0, st->cubeVbh);
        bgfx::setIndexBuffer(st->cubeIbh);
        bgfx::setInstanceDataBuffer(&idb);
        bgfx::setState(BGFX_STATE_DEFAULT);
        bgfx::submit(0, st->program);
    }

    bgfx::frame();
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (bgfx::isValid(st->cubeIbh)) bgfx::destroy(st->cubeIbh);
    if (bgfx::isValid(st->cubeVbh)) bgfx::destroy(st->cubeVbh);
    if (bgfx::isValid(st->program)) bgfx::destroy(st->program);
    bgfx::shutdown();
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Press your stats key from recipe 02 (or set BGFX_DEBUG_STATS). The draw count stays at 1 while 121 cubes spin — the whole point of instancing.
  2. Bump kSide to 40 (1 600 cubes). Still one draw call; watch the GPU time, not the CPU submit count, rise.
  3. Widen the stride to 80 and append a per-instance colour vec4 as i_data4 (declare it in varying.def.sc). Multiply v_color0 by it to tint each cube individually.
  4. Remove the getAvailInstanceDataBuffer guard and request a million instances. bgfx clamps to its scratch budget — the guard is how you'd notice and draw a sensible subset.
