Recipe 07 — The Spinning Cube

8 vertices, 36 indices, a time-driven rotation — your first real 3D object. And, on purpose, your first look at a 3D scene rendered without depth.

What you'll learn

  • The canonical bgfx cube: 8 corner vertices reused across 12 triangles via a 36-entry index buffer.
  • Animating the model matrix each frame with bx::mtxRotateXY.
  • Pulling the camera back so a 2-unit cube fits the frame.
  • Why the cube looks scrambled with no depth buffer — the motivation for recipe 08.

Prerequisites

Recipe 06 — the MVP via setTransform + setViewTransform and u_modelViewProj. We reuse exactly that machinery; only the geometry grows from a quad to a cube. Shaders here are a plain position+colour pair, vs_cubes / fs_cubes (recipes 08, 10 and 11 reuse them).

Walkthrough

Eight corners, twelve triangles

A cube has 8 corners. Each of its 6 faces is 2 triangles, so 12 triangles × 3 = 36 indices — all pointing back into those same 8 vertices. This is the payoff of recipe 05's index buffer at a meaningful scale.

static const PosColorVertex kCube[8] = {

    0, 1, 2,  1, 3, 2,    // +Z face

    2, 3, 6,  6, 3, 7,    // -Y face

Spinning it

Identical camera setup to recipe 06, just pulled further back so the 2-unit cube fits, and a two-axis rotation that updates every frame. 

const bx::Vec3 at  = { 0.0f, 0.0f,  0.0f };
const bx::Vec3 eye = { 0.0f, 0.0f, -8.0f };
float view[16], proj[16];
bx::mtxLookAt(view, eye, at);
bx::mtxProj(proj, 60.0f, float(st->width)/float(st->height), 0.1f, 100.0f,
            bgfx::getCaps()->homogeneousDepth);
bgfx::setViewTransform(0, view, proj);

float model[16];
bx::mtxRotateXY(model, t * 0.21f, t * 0.37f);   // tumble on X and Y
bgfx::setTransform(model);
Expect a mess. We submit the cube with only colour writes — no depth test, no depth clear. The GPU draws triangles in index order, so faces that should be hidden behind nearer ones paint right over them. The cube's silhouette tumbles correctly, but its surface flickers and turns inside-out. That's not a bug — it's exactly the problem recipe 08 solves with one depth flag and one depth clear.

Complete listing

/*
 * bgfx/07-spinning-cube.cpp — a tumbling RGB cube, deliberately WITHOUT depth.
 * Native-handle + loadProgram helpers are identical to recipe 03 (elided).
 * Shaders: vs_cubes / fs_cubes (position + color, MVP via u_modelViewProj).
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include <bgfx/bgfx.h>
#include <bgfx/platform.h>
#include <bx/math.h>
#include <cstdint>

static void *nativeWindowHandle(SDL_Window *);
static void *nativeDisplayHandle(SDL_Window *);
static bgfx::ProgramHandle loadProgram(const char *vs, const char *fs);

struct PosColorVertex { float x, y, z; uint32_t abgr; };
static const PosColorVertex kCube[8] = {
    { -1.0f,  1.0f,  1.0f, 0xff000000 }, {  1.0f,  1.0f,  1.0f, 0xff0000ff },
    { -1.0f, -1.0f,  1.0f, 0xff00ff00 }, {  1.0f, -1.0f,  1.0f, 0xff00ffff },
    { -1.0f,  1.0f, -1.0f, 0xffff0000 }, {  1.0f,  1.0f, -1.0f, 0xffff00ff },
    { -1.0f, -1.0f, -1.0f, 0xffffff00 }, {  1.0f, -1.0f, -1.0f, 0xffffffff },
};
static const uint16_t kCubeIdx[36] = {
    0,1,2, 1,3,2,  4,6,5, 5,6,7,  0,2,4, 4,2,6,
    1,5,3, 5,7,3,  0,4,1, 4,5,1,  2,3,6, 6,3,7,
};

struct AppState {
    SDL_Window              *window = nullptr;
    int                      width = 1280, height = 720;
    bgfx::VertexLayout       layout;
    bgfx::VertexBufferHandle vbh = BGFX_INVALID_HANDLE;
    bgfx::IndexBufferHandle  ibh = BGFX_INVALID_HANDLE;
    bgfx::ProgramHandle      program = BGFX_INVALID_HANDLE;
    Uint64                   startNs = 0;
};

static SDL_AppResult fail(const char *what) { SDL_Log("%s: %s", what, SDL_GetError()); return SDL_APP_FAILURE; }

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("bgfx Cube", "1.0", "com.example.sdl3.bgfx.cube");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("SDL_Init");

    AppState *st = new AppState(); *appstate = st;
    st->window = SDL_CreateWindow("bgfx — The Spinning Cube", st->width, st->height,
                                  SDL_WINDOW_RESIZABLE | SDL_WINDOW_HIGH_PIXEL_DENSITY);
    if (!st->window) return fail("SDL_CreateWindow");
    SDL_GetWindowSizeInPixels(st->window, &st->width, &st->height);

    bgfx::renderFrame();
    bgfx::Init init;
    init.type              = bgfx::RendererType::Count;
    init.platformData.nwh  = nativeWindowHandle(st->window);
    init.platformData.ndt  = nativeDisplayHandle(st->window);
    init.resolution.width  = (uint32_t)st->width;
    init.resolution.height = (uint32_t)st->height;
    init.resolution.reset  = BGFX_RESET_VSYNC;
    if (!bgfx::init(init)) return fail("bgfx::init");

    bgfx::setViewClear(0, BGFX_CLEAR_COLOR, 0x121a30ff, 1.0f, 0);   // no DEPTH clear (yet)

    st->layout.begin()
        .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
        .add(bgfx::Attrib::Color0,   4, bgfx::AttribType::Uint8, true)
        .end();
    st->vbh     = bgfx::createVertexBuffer(bgfx::makeRef(kCube, sizeof(kCube)), st->layout);
    st->ibh     = bgfx::createIndexBuffer (bgfx::makeRef(kCubeIdx, sizeof(kCubeIdx)));
    st->program = loadProgram("vs_cubes", "fs_cubes");
    if (!bgfx::isValid(st->program)) return fail("loadProgram");

    st->startNs = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_WINDOW_PIXEL_SIZE_CHANGED) {
        st->width = e->window.data1; st->height = e->window.data2;
        bgfx::reset((uint32_t)st->width, (uint32_t)st->height, BGFX_RESET_VSYNC);
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (float)(SDL_GetTicksNS() - st->startNs) / 1.0e9f;

    bgfx::setViewRect(0, 0, 0, (uint16_t)st->width, (uint16_t)st->height);

    const bx::Vec3 at  = { 0.0f, 0.0f,  0.0f };
    const bx::Vec3 eye = { 0.0f, 0.0f, -8.0f };
    float view[16], proj[16];
    bx::mtxLookAt(view, eye, at);
    bx::mtxProj(proj, 60.0f, float(st->width)/float(st->height), 0.1f, 100.0f,
                bgfx::getCaps()->homogeneousDepth);
    bgfx::setViewTransform(0, view, proj);

    float model[16];
    bx::mtxRotateXY(model, t * 0.21f, t * 0.37f);
    bgfx::setTransform(model);

    bgfx::setVertexBuffer(0, st->vbh);
    bgfx::setIndexBuffer(st->ibh);
    bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);   // no depth — see recipe 08
    bgfx::submit(0, st->program);

    bgfx::frame();
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (bgfx::isValid(st->ibh))     bgfx::destroy(st->ibh);
    if (bgfx::isValid(st->vbh))     bgfx::destroy(st->vbh);
    if (bgfx::isValid(st->program)) bgfx::destroy(st->program);
    bgfx::shutdown();
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Slow the spin right down (multiply t by 0.05). With the cube nearly still, the depth problem is easier to see — watch a far face bleed through a near one.
  2. Rotate on a single axis (bx::mtxRotateY(model, t * 0.3f)). The artefacts change character but never disappear — order, not motion, is the issue.
  3. Dolly the camera to eye = { 0, 0, -4 }. The cube nearly fills the frame and may clip the near plane (0.1) when a corner swings close.
  4. Peek ahead: add | BGFX_STATE_WRITE_Z | BGFX_STATE_DEPTH_TEST_LESS to setState and | BGFX_CLEAR_DEPTH to the clear. The cube snaps solid — recipe 08 explains exactly why.
