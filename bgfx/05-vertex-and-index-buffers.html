Recipe 05 — Vertex & Index Buffers

Upload geometry to the GPU once with static buffers, then draw a quad from 4 vertices and 6 indices instead of 6 duplicated vertices.

What you'll learn

  • Static createVertexBuffer / createIndexBuffer — uploaded in init, kept for the program's life.
  • bgfx::makeRef vs bgfx::copy: who owns the bytes, and when each is safe.
  • Why an index buffer lets 4 vertices cover a quad's 2 triangles.
  • The draw call with persistent handles, and freeing them in SDL_AppQuit.

Prerequisites

The triangle and its vs_triangle/fs_triangle program from recipe 03 — the quad reuses them unchanged. Recipe 03's transient buffer was per-frame scratch; here we go the other way.

Walkthrough

Transient vs. static

Recipe 03's transient buffer is reallocated and refilled every frame — right for data that changes constantly, wasteful for geometry that doesn't. A quad's corners never move, so we upload them once into a static VertexBufferHandle and just reference it each frame.

makeRef vs copy

Both hand bgfx a bgfx::Memory block. bgfx::copy duplicates your bytes immediately — safe always, costs a copy. bgfx::makeRef aliases your bytes with no copy, but you must keep them alive until bgfx is done. For a static const array that lives as long as the program, makeRef is free and correct. 

struct PosColorVertex { float x, y, z; uint32_t abgr; };  // ABGR, see recipe 03

static const PosColorVertex kQuad[4] = {
    { -0.5f,  0.5f, 0.0f, 0xff0000ff },   // 0  top-left   red
    {  0.5f,  0.5f, 0.0f, 0xff00ff00 },   // 1  top-right  green
    {  0.5f, -0.5f, 0.0f, 0xffff0000 },   // 2  bot-right  blue
    { -0.5f, -0.5f, 0.0f, 0xff00ffff },   // 3  bot-left   yellow
};
static const uint16_t kIndices[6] = { 0, 1, 2,   0, 2, 3 };   // two triangles

Creating the buffers

The vertex buffer needs the same VertexLayout from recipe 03. Indices are 16-bit by default (enough for 65 536 vertices); pass BGFX_BUFFER_INDEX32 if you need more. 

st->vbh = bgfx::createVertexBuffer(bgfx::makeRef(kQuad, sizeof(kQuad)), st->layout);
st->ibh = bgfx::createIndexBuffer (bgfx::makeRef(kIndices, sizeof(kIndices)));

The two shared triangles meet along the 0–2 diagonal. Without indices we'd repeat vertices 0 and 2 — fine for a quad, but the saving compounds fast: the cube in recipe 07 reuses its 8 corners across 12 triangles.

Drawing

Per frame, bind both buffers and submit. setIndexBuffer is the only addition over recipe 03's draw. 

bgfx::setVertexBuffer(0, st->vbh);
bgfx::setIndexBuffer(st->ibh);
bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);
bgfx::submit(0, st->program);

Static handles must be released explicitly — bgfx won't garbage-collect them — so SDL_AppQuit calls bgfx::destroy on each before shutdown.

Complete listing

/*
 * bgfx/05-vertex-and-index-buffers.cpp — a static, indexed quad.
 * Native-handle + loadProgram helpers are identical to recipe 03.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include <bgfx/bgfx.h>
#include <bgfx/platform.h>
#include <cstdint>

static void *nativeWindowHandle(SDL_Window *w) {
    SDL_PropertiesID p = SDL_GetWindowProperties(w);
#if defined(SDL_PLATFORM_MACOS)
    return SDL_GetPointerProperty(p, SDL_PROP_WINDOW_COCOA_WINDOW_POINTER, nullptr);
#elif defined(SDL_PLATFORM_WIN32)
    return SDL_GetPointerProperty(p, SDL_PROP_WINDOW_WIN32_HWND_POINTER, nullptr);
#elif defined(SDL_PLATFORM_LINUX)
    if (SDL_strcmp(SDL_GetCurrentVideoDriver(), "wayland") == 0)
        return SDL_GetPointerProperty(p, SDL_PROP_WINDOW_WAYLAND_SURFACE_POINTER, nullptr);
    return (void *)(uintptr_t)SDL_GetNumberProperty(p, SDL_PROP_WINDOW_X11_WINDOW_NUMBER, 0);
#else
    return nullptr;
#endif
}
static void *nativeDisplayHandle(SDL_Window *w) {
#if defined(SDL_PLATFORM_LINUX)
    SDL_PropertiesID p = SDL_GetWindowProperties(w);
    if (SDL_strcmp(SDL_GetCurrentVideoDriver(), "wayland") == 0)
        return SDL_GetPointerProperty(p, SDL_PROP_WINDOW_WAYLAND_DISPLAY_POINTER, nullptr);
    return SDL_GetPointerProperty(p, SDL_PROP_WINDOW_X11_DISPLAY_POINTER, nullptr);
#else
    (void)w; return nullptr;
#endif
}
static const bgfx::Memory *loadMem(const char *path) {
    size_t size = 0; void *data = SDL_LoadFile(path, &size);
    if (!data) { SDL_Log("missing shader: %s", path); return nullptr; }
    const bgfx::Memory *mem = bgfx::copy(data, (uint32_t)size); SDL_free(data); return mem;
}
static const char *shaderDir() {
    switch (bgfx::getRendererType()) {
        case bgfx::RendererType::Metal:      return "shaders/metal/";
        case bgfx::RendererType::Vulkan:     return "shaders/spirv/";
        case bgfx::RendererType::Direct3D11:
        case bgfx::RendererType::Direct3D12: return "shaders/dx11/";
        case bgfx::RendererType::OpenGL:     return "shaders/glsl/";
        default:                             return "shaders/";
    }
}
static bgfx::ProgramHandle loadProgram(const char *vs, const char *fs) {
    char path[512];
    SDL_snprintf(path, sizeof(path), "%s%s.bin", shaderDir(), vs);
    bgfx::ShaderHandle vsh = bgfx::createShader(loadMem(path));
    SDL_snprintf(path, sizeof(path), "%s%s.bin", shaderDir(), fs);
    bgfx::ShaderHandle fsh = bgfx::createShader(loadMem(path));
    return bgfx::createProgram(vsh, fsh, true);
}

struct PosColorVertex { float x, y, z; uint32_t abgr; };
static const PosColorVertex kQuad[4] = {
    { -0.5f,  0.5f, 0.0f, 0xff0000ff },
    {  0.5f,  0.5f, 0.0f, 0xff00ff00 },
    {  0.5f, -0.5f, 0.0f, 0xffff0000 },
    { -0.5f, -0.5f, 0.0f, 0xff00ffff },
};
static const uint16_t kIndices[6] = { 0, 1, 2,  0, 2, 3 };

struct AppState {
    SDL_Window              *window = nullptr;
    int                      width = 1280, height = 720;
    bgfx::VertexLayout       layout;
    bgfx::VertexBufferHandle vbh = BGFX_INVALID_HANDLE;
    bgfx::IndexBufferHandle  ibh = BGFX_INVALID_HANDLE;
    bgfx::ProgramHandle      program = BGFX_INVALID_HANDLE;
};

static SDL_AppResult fail(const char *what) { SDL_Log("%s: %s", what, SDL_GetError()); return SDL_APP_FAILURE; }

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("bgfx Quad", "1.0", "com.example.sdl3.bgfx.quad");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("SDL_Init");

    AppState *st = new AppState(); *appstate = st;
    st->window = SDL_CreateWindow("bgfx — Vertex & Index Buffers", st->width, st->height,
                                  SDL_WINDOW_RESIZABLE | SDL_WINDOW_HIGH_PIXEL_DENSITY);
    if (!st->window) return fail("SDL_CreateWindow");
    SDL_GetWindowSizeInPixels(st->window, &st->width, &st->height);

    bgfx::renderFrame();
    bgfx::Init init;
    init.type              = bgfx::RendererType::Count;
    init.platformData.nwh  = nativeWindowHandle(st->window);
    init.platformData.ndt  = nativeDisplayHandle(st->window);
    init.resolution.width  = (uint32_t)st->width;
    init.resolution.height = (uint32_t)st->height;
    init.resolution.reset  = BGFX_RESET_VSYNC;
    if (!bgfx::init(init)) return fail("bgfx::init");

    bgfx::setViewClear(0, BGFX_CLEAR_COLOR, 0x121a30ff, 1.0f, 0);
    bgfx::setViewRect(0, 0, 0, (uint16_t)st->width, (uint16_t)st->height);

    st->layout.begin()
        .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
        .add(bgfx::Attrib::Color0,   4, bgfx::AttribType::Uint8, true)
        .end();

    st->vbh     = bgfx::createVertexBuffer(bgfx::makeRef(kQuad, sizeof(kQuad)), st->layout);
    st->ibh     = bgfx::createIndexBuffer (bgfx::makeRef(kIndices, sizeof(kIndices)));
    st->program = loadProgram("vs_triangle", "fs_triangle");
    if (!bgfx::isValid(st->program)) return fail("loadProgram");

    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_WINDOW_PIXEL_SIZE_CHANGED) {
        st->width = e->window.data1; st->height = e->window.data2;
        bgfx::reset((uint32_t)st->width, (uint32_t)st->height, BGFX_RESET_VSYNC);
        bgfx::setViewRect(0, 0, 0, (uint16_t)st->width, (uint16_t)st->height);
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    bgfx::setVertexBuffer(0, st->vbh);
    bgfx::setIndexBuffer(st->ibh);
    bgfx::setState(BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A);
    bgfx::submit(0, st->program);
    bgfx::frame();
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (bgfx::isValid(st->ibh))     bgfx::destroy(st->ibh);
    if (bgfx::isValid(st->vbh))     bgfx::destroy(st->vbh);
    if (bgfx::isValid(st->program)) bgfx::destroy(st->program);
    bgfx::shutdown();
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Drop the index buffer: expand kQuad to 6 vertices (repeat 0 and 2) and call setVertexBuffer without an index buffer. Same picture, more data — the case indices exist to avoid.
  2. Change the indices to { 0, 1, 2 } and draw only the first triangle. The index buffer, not the vertex buffer, decides what's drawn.
  3. Swap makeRef for copy. Identical result; now bgfx owns its own copy, so the source array could safely be a local you let go out of scope.
  4. Comment out the two bgfx::destroy calls and run with a debug bgfx build — it logs leaked handles at shutdown. Proof static handles aren't auto-freed.
